    .then(resp => resp.text())
    .then(text => {
        if (text.indexOf('html')>=0) {
            document.querySelector('#notif-content').innerText = '?'
            document.querySelector('#notif-content').style.color = '#3B8FD2'
        } else {
            document.querySelector('#notif-content').innerText = text
            document.querySelector('#notif-content').style.color = text == '0' ? '#3B8FD2' : '#DD4B39'
        }
    })
