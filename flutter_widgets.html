    child: Image.asset('assets/images/logo-mikajy.png', width: 100, height: 100),
)

    onTap: () {
        setState(() {
            pageMode = PAGE_MODE_SUBSCRIBE;
        }),
    },
    child: Image.asset('assets/images/logo-mikajy.png'),
)

//Largeur et hauteur de l'écran
double width = MediaQuery.sizeOf(context).width;
double height = MediaQuery.sizeOf(context).height;

//Largeur et hauteur du safe area (zone manipulable) de l'écran
var padding = MediaQuery.paddingOf(context);
double newheight = height - padding.top - padding.bottom;

//Sans contexte
// First get the FlutterView
FlutterView view = WidgetsBinding.instance.platformDispatcher.views.first;
// Dimensions in physical pixels (px)
Size size = view.physicalSize;
double width = size.width;
double height = size.height;
// Dimensions in logical pixels (dp)
Size size = view.physicalSize / view.devicePixelRatio;
double width = size.width;
double height = size.height;
